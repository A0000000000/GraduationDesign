# 用户模型
* id: 用户id, String
* username: 用户名, String
* password: 密码, String
* email: 邮箱, String
* createTime: 创建日期, Date
* birthday: 生日, Date
* fileBucket: 文件桶, String
* icon: 头像, String
* role: 权限, Array<String>
* locations: 收货地址, Array<String>
* orders: 订单, Array<String>
* couponItem: 所持有的优惠券, Array<String>
* status: 用户状态, String

# 收货地址模型
* id: 收货地址id, String
* userId: 创建人id, String
* name: 收货地址, String
* details: 详细地址, String

# 优惠券模型
* id: 优惠券id, String
* userId: 创建优惠券的运维, String
* sum: 优惠券总量, Number
* count: 优惠量, Number
* price: 优惠券限制, Number
* good: 限制使用商品, Array<String>

# 用户持有优惠券的模型
* id: 模型id, String
* couponItemId: 优惠券id, String
* count: 持有数量, Number
* userId: 持有人, String

# 订单模型
* id: 模型id, String
* userId: 创建订单的用户, String
* orderItem: 订单条目, Array<String>
* status: 订单状态, String
* createTime: 创建时间, Date
* sum: 待支付金额, Number

# 条目模型
* id: 条目id, String
* goodId: 商品id, String
* count: 商品数量
* coupons: 所使用的优惠券id, Array<String>

# 商品模型
* id: 商品id, String
* price: 商品价格, Number
* count: 商品库存, Number
* name: 商品名称, String
* fileBucket: 商品图片文件桶, String
* files: 商品图片, Array<String>
* details: 商品介绍, String
* type: 商品类型, String


# 活动模型
* id: 活动id, String
* userId: 发起人, String,
* fileBucket: 活动图片文件桶, String
* files: 活动图片, Array<String>
* couponId: 活动发放的优惠券, String
* title: 活动标题, String
* content: 活动内容, String
